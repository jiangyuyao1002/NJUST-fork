327 | 		try {
328 | 			await this.taskHistoryStore.initialize()
329 | 
330 | 			// Migration: backfill per-task files from globalState on first run
331 | 			const migrationKey = "taskHistoryMigratedToFiles"
332 | 			const alreadyMigrated = this.context.globalState.get<boolean>(migrationKey)
333 | 
334 | 			if (!alreadyMigrated) {
335 | 				const legacyHistory = this.context.globalState.get<HistoryItem[]>("taskHistory") ?? []
336 | 
337 | 				if (legacyHistory.length > 0) {
338 | 					this.log(`[initializeTaskHistoryStore] Migrating ${legacyHistory.length} entries from globalState`)
339 | 					await this.taskHistoryStore.migrateFromGlobalState(legacyHistory)
340 | 				}
341 | 
342 | 				await this.context.globalState.update(migrationKey, true)
343 | 				this.log("[initializeTaskHistoryStore] Migration complete")
344 | 			}
345 |

347 | 
348 | 			try {

350 | 					if (id === NO_TASK_KEY) return false
351 | 					const h = this.taskHistoryStore.get(id)
352 | 					if (!h || h.status === "completed") return false
353 | 					return true
354 | 				})
355 | 				if (filesRemoved > 0) {
356 | 					this.outputChannel.appendLine(
357 | 						`[CangjieTestCleanup] 启动修剪：移除 ${filesRemoved} 个生成测试文件（${taskEntriesRemoved} 个任务桶）。`,
358 | 					)

360 | 			} catch (e) {

362 | 			}
363 | 
364 | 			// Push task history to webview now that it's ready,
365 | 			// so the UI that loaded with an empty list gets updated.

434 | 	// Removes and destroys the top Cline instance (the current finished task),
435 | 	// activating the previous one (resuming the parent task).
436 | 	async removeClineFromStack(options?: { skipDelegationRepair?: boolean }) {
437 | 		if (this.clineStack.length === 0) {
438 | 			return
439 | 		}
440 | 
441 | 		// Pop the top Cline instance from the stack.
442 | 		let task = this.clineStack.pop()
443 | 
444 | 		if (task) {
445 | 			// Capture delegation metadata before abort/dispose, since abortTask(true)
446 | 			// is async and the task reference is cleared afterwards.
447 | 			const childTaskId = task.taskId
448 | 			const parentTaskId = task.parentTaskId
449 | 
450 | 			task.emit(NJUST_AIEventName.TaskUnfocused)
451 | 
452 | 			try {
453 | 				// Abort the running task and set isAbandoned to true so
454 | 				// all running promises will exit as well.
455 | 				await task.abortTask(true)
456 | 			} catch (e) {
457 | 				this.log(
458 | 					`[ClineProvider#removeClineFromStack] abortTask() failed ${task.taskId}.${task.instanceId}: ${e.message}`,
459 | 				)
460 | 			}
461 | 
462 | 			// Remove event listeners before clearing the reference.
463 | 			const cleanupFunctions = this.taskEventListeners.get(task)
464 | 
465 | 			if (cleanupFunctions) {
466 | 				cleanupFunctions.forEach((cleanup) => cleanup())
467 | 				this.taskEventListeners.delete(task)
468 | 			}
469 | 
470 | 			// Make sure no reference kept, once promises end it will be
471 | 			// garbage collected.
472 | 			task = undefined
473 | 
474 | 			// Delegation-aware parent metadata repair:
475 | 			// If the popped task was a delegated child, repair the parent's metadata
476 | 			// so it transitions from "delegated" back to "active" and becomes resumable
477 | 			// from the task history list.
478 | 			// Skip when called from delegateParentAndOpenChild() during nested delegation
479 | 			// transitions (A→B→C), where the caller intentionally replaces the active
480 | 			// child and will update the parent to point at the new child.

481 | 			if (parentTaskId && childTaskId && !options?.skipDelegationRepair) {
482 | 				try {
483 | 					const { historyItem: parentHistory } = await this.getTaskWithId(parentTaskId)
484 | 
485 | 					if (parentHistory.status === "delegated" && parentHistory.awaitingChildId === childTaskId) {
486 | 						await this.updateTaskHistory({
487 | 							...parentHistory,
488 | 							status: "active",
489 | 							awaitingChildId: undefined,
490 | 						})
491 | 						this.log(
492 | 							`[ClineProvider#removeClineFromStack] Repaired parent ${parentTaskId} metadata: delegated → active (child ${childTaskId} removed)`,
493 | 						)
