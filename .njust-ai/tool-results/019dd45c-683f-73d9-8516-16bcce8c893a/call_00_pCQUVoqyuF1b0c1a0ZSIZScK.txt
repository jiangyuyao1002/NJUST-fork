327 | 		try {
328 | 			await this.taskHistoryStore.initialize()
329 | 
330 | 			// Migration: backfill per-task files from globalState on first run
331 | 			const migrationKey = "taskHistoryMigratedToFiles"
332 | 			const alreadyMigrated = this.context.globalState.get<boolean>(migrationKey)
333 | 
334 | 			if (!alreadyMigrated) {
335 | 				const legacyHistory = this.context.globalState.get<HistoryItem[]>("taskHistory") ?? []
336 | 
337 | 				if (legacyHistory.length > 0) {
338 | 					this.log(`[initializeTaskHistoryStore] Migrating ${legacyHistory.length} entries from globalState`)
339 | 					await this.taskHistoryStore.migrateFromGlobalState(legacyHistory)
340 | 				}
341 | 
342 | 				await this.context.globalState.update(migrationKey, true)
343 | 				this.log("[initializeTaskHistoryStore] Migration complete")
344 | 			}
345 | 
346 | 			this.taskHistoryStoreInitialized = true
347 | 
348 | 			try {
349 | 				const { filesRemoved, taskEntriesRemoved } = pruneStaleRegistrations((id) => {
350 | 					if (id === NO_TASK_KEY) return false
351 | 					const h = this.taskHistoryStore.get(id)
352 | 					if (!h || h.status === "completed") return false
353 | 					return true
354 | 				})
355 | 				if (filesRemoved > 0) {
356 | 					this.outputChannel.appendLine(
357 | 						`[CangjieTestCleanup] 启动修剪：移除 ${filesRemoved} 个生成测试文件（${taskEntriesRemoved} 个任务桶）。`,
358 | 					)
359 | 				}
360 | 			} catch (e) {
361 | 				this.log(`[CangjieTestCleanup] prune failed: ${e instanceof Error ? e.message : String(e)}`)
362 | 			}
363 | 
364 | 			// Push task history to webview now that it's ready,
365 | 			// so the UI that loaded with an empty list gets updated.
366 | 			const items = this.taskHistoryStore.getAll().filter((item: HistoryItem) => item.ts && item.task)
367 | 			this.postMessageToWebview({ type: "taskHistoryUpdated", taskHistory: items })
368 | 		} catch (error) {

370 | 		}

435 | 	// Removes and destroys the top Cline instance (the current finished task),
436 | 	// activating the previous one (resuming the parent task).
437 | 	async removeClineFromStack(options?: { skipDelegationRepair?: boolean }) {
438 | 		if (this.clineStack.length === 0) {
439 | 			return
440 | 		}
441 | 
442 | 		// Pop the top Cline instance from the stack.
443 | 		let task = this.clineStack.pop()
444 | 
445 | 		if (task) {
446 | 			// Capture delegation metadata before abort/dispose, since abortTask(true)
447 | 			// is async and the task reference is cleared afterwards.
448 | 			const childTaskId = task.taskId
449 | 			const parentTaskId = task.parentTaskId
450 | 
451 | 			task.emit(NJUST_AIEventName.TaskUnfocused)
452 | 
453 | 			try {
454 | 				// Abort the running task and set isAbandoned to true so
455 | 				// all running promises will exit as well.
456 | 				await task.abortTask(true)
457 | 			} catch (e) {
458 | 				const msg = e instanceof Error ? e.message : String(e)
459 | 				this.log(
460 | 					`[ClineProvider#removeClineFromStack] abortTask() failed ${task.taskId}.${task.instanceId}: ${msg}`,
461 | 				)
462 | 			}
463 | 
464 | 			// Remove event listeners before clearing the reference.
465 | 			const cleanupFunctions = this.taskEventListeners.get(task)
466 | 
467 | 			if (cleanupFunctions) {
468 | 				cleanupFunctions.forEach((cleanup) => cleanup())
469 | 				this.taskEventListeners.delete(task)
470 | 			}
471 | 
472 | 			// Make sure no reference kept, once promises end it will be
473 | 			// garbage collected.
474 | 			task = undefined
475 | 
476 | 			// Delegation-aware parent metadata repair:
477 | 			// If the popped task was a delegated child, repair the parent's metadata
478 | 			// so it transitions from "delegated" back to "active" and becomes resumable
479 | 			// from the task history list.
480 | 			// Skip when called from delegateParentAndOpenChild() during nested delegation
481 | 			// transitions (A→B→C), where the caller intentionally replaces the active
482 | 			// child and will update the parent to point at the new child.
483 | 			if (parentTaskId && childTaskId && !options?.skipDelegationRepair) {
484 | 				try {
485 | 					const { historyItem: parentHistory } = await this.getTaskWithId(parentTaskId)
486 | 
487 | 					if (parentHistory.status === "delegated" && parentHistory.awaitingChildId === childTaskId) {
488 | 						await this.updateTaskHistory({
489 | 							...parentHistory,
490 | 							status: "active",
491 | 							awaitingChildId: undefined,
