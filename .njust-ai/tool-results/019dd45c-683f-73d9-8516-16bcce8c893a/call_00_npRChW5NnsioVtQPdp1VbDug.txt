196 | /**
197 |  * Read actual `package` declarations from .cj source files and compare
198 |  * with directory-inferred package names. Report mismatches so the AI
199 |  * can generate correct package declarations.
200 |  */
201 | function verifyPackageDeclarations(
202 | 	root: PackageNode,
203 | 	cwd: string,
204 | 	srcDir: string,
205 | ): string | null {
206 | 	const mismatches: string[] = []
207 | 	const MAX_CHECKS = 50
208 | 	let checked = 0
209 | 	const symbolIndex = CangjieSymbolIndex.getInstance()
210 | 
211 | 	function walk(node: PackageNode): void {
212 | 		if (checked >= MAX_CHECKS) return
213 | 
214 | 		for (const fileName of node.sourceFiles) {
215 | 			if (checked >= MAX_CHECKS) return
216 | 			checked++
217 | 
218 | 			const filePath = path.join(cwd, node.dirPath, fileName)
219 | 			const expectedPkg = node.packageName
220 | 			let declaredPkg: string | null = null
221 | 
222 | 			// Prefer SymbolIndex (no file I/O) over fs.readFileSync
223 | 			if (symbolIndex) {
224 | 				const syms = symbolIndex.getSymbolsByFile(filePath)
225 | 				const pkgSym = syms.find((s) => s.kind === "package")
226 | 				if (pkgSym) declaredPkg = pkgSym.name
227 | 			}
228 | 
229 | 			// Fallback: read from disk (SymbolIndex may not include package entries)
230 | 			if (declaredPkg === null) {
231 | 				try {
232 | 					const content = fs.readFileSync(filePath, "utf-8")
233 | 					const match = content.match(PACKAGE_DECL_REGEX)
234 | 					declaredPkg = match ? match[1] : null
235 | 				} catch {
236 | 					continue
237 | 				}
238 | 			}
239 | 
240 | 			if (declaredPkg && declaredPkg !== expectedPkg) {

241 | 				const relPath = path.relative(cwd, filePath).replace(/\\/g, "/")
242 | 				mismatches.push(
243 | 					`- ${relPath}: 声明 \`package ${declaredPkg}\`，但目录推导应为 \`package ${expectedPkg}\``,
244 | 				)
245 | 			} else if (!declaredPkg && node.packageName.includes(".")) {
246 | 				const relPath = path.relative(cwd, filePath).replace(/\\/g, "/")
247 | 				mismatches.push(
248 | 					`- ${relPath}: **缺少 package 声明**，应声明 \`package ${expectedPkg}\``,
249 | 				)
250 | 			}
251 | 		}
252 | 
253 | 		for (const child of node.children) {
254 | 			walk(child)
255 | 		}
256 | 	}
257 | 
258 | 	walk(root)
259 | 
260 | 	if (mismatches.length === 0) return null
261 | 
262 | 	return (
263 | 		`## ⚠ 包声明不一致\n\n` +
264 | 		`以下文件的 \`package\` 声明与目录结构不匹配，**生成代码时请使用正确的包名**：\n\n` +
265 | 		mismatches.join("\n") +
266 | 		`\n\n规则: 文件所在目录相对于 ${srcDir}/ 的路径决定包名（如 ${srcDir}/network/http/ → package <root>.network.http）`
267 | 	)
268 | }

286 | 
287 | 	const MAX_SYMBOLS_PER_MODULE = 20

289 | 
290 | 	for (const member of info.members) {
291 | 		const memberSrcDir = path.join(cwd, member.path, "src")

295 | 		if (symbols.length === 0) continue

308 | 		})

391 | function sampleCangjieDiagnostics(
392 | 	diags: vscode.Diagnostic[],
393 | 	opts?: { maxErrors?: number; maxWarnings?: number },
394 | ): { sampled: vscode.Diagnostic[]; total: number; omitted: number } {
395 | 	const maxE = opts?.maxErrors ?? DIAG_SAMPLE_MAX_ERRORS
396 | 	const maxW = opts?.maxWarnings ?? DIAG_SAMPLE_MAX_WARNINGS
397 | 	const total = diags.length
398 | 
399 | 	const active = getActiveCangjieFileInfo()
400 | 	const activeUri = vscode.window.activeTextEditor?.document.uri.toString()
401 | 	const normMemo = new Map<string, string>()
402 | 	const normAgg = (msg: string) => {
403 | 		let v = normMemo.get(msg)
404 | 		if (v === undefined) {
405 | 			v = normalizeDiagnosticMessageForAggregation(msg)
406 | 			normMemo.set(msg, v)
407 | 		}
408 | 		return v
409 | 	}
410 | 	const byBucket = new Map<string, vscode.Diagnostic[]>()
411 | 	for (const d of diags) {
412 | 		const code = normalizeDiagnosticCode(d) ?? "-"
413 | 		const key = `${d.severity}-${code}-${normAgg(d.message)}`
414 | 		const arr = byBucket.get(key)
415 | 		if (arr) arr.push(d)
416 | 		else byBucket.set(key, [d])
417 | 	}
418 | 	const patternCache = buildDiagnosticPatternCache(diags)
419 | 
420 | 	const sevRank = (s: vscode.DiagnosticSeverity) =>

421 | 		s === vscode.DiagnosticSeverity.Error ? 0 : s === vscode.DiagnosticSeverity.Warning ? 1 : 2
422 | 
423 | 	const patternPri = (d: vscode.Diagnostic) => patternCache.get(d)?.priority ?? 0
424 | 
425 | 	const bucketsWithMin = [...byBucket.values()].map((group) => ({
426 | 		group,
427 | 		minLine: group.reduce((m, x) => Math.min(m, x.range.start.line), Number.MAX_SAFE_INTEGER),
428 | 	}))
429 | 	bucketsWithMin.sort((ba, bb) => {
430 | 		const a = ba.group
431 | 		const b = bb.group
432 | 		const da = a[0]
433 | 		const db = b[0]
434 | 		const r = sevRank(da.severity) - sevRank(db.severity)
435 | 		if (r !== 0) return r
436 | 		const pa = patternPri(da)
437 | 		const pb = patternPri(db)
438 | 		if (Math.abs(pa - pb) >= 10) return pb - pa
439 | 		const pr = pb - pa
440 | 		if (pr !== 0) return pr
441 | 		const la = ba.minLine
442 | 		const lb = bb.minLine
443 | 		if (!active || !activeUri) return la - lb
444 | 		const uriA = DIAGNOSTIC_URI_MAP.get(da)
445 | 		const uriB = DIAGNOSTIC_URI_MAP.get(db)
446 | 		const distA = uriA === activeUri ? Math.abs(la - active.cursorLine) : Number.MAX_SAFE_INTEGER / 2 + la
447 | 		const distB = uriB === activeUri ? Math.abs(lb - active.cursorLine) : Number.MAX_SAFE_INTEGER / 2 + lb
448 | 		return distA - distB
449 | 	})
450 | 	const buckets = bucketsWithMin.map((x) => x.group)
451 | 
452 | 	const sampled: vscode.Diagnostic[] = []
453 | 	let errTaken = 0
454 | 	let warnTaken = 0
455 | 	let covered = 0
456 | 
457 | 	for (const group of buckets) {
458 | 		const rep = group[0]
459 | 		if (rep.severity === vscode.DiagnosticSeverity.Error) {
460 | 			if (errTaken >= maxE) continue
461 | 			errTaken++
462 | 		} else if (rep.severity === vscode.DiagnosticSeverity.Warning) {
463 | 			if (warnTaken >= maxW) continue
464 | 			warnTaken++
465 | 		} else {
466 | 			continue
467 | 		}
468 | 
469 | 		covered += group.length
470 | 		if (group.length === 1) {
471 | 			sampled.push(rep)
472 | 		} else {
473 | 			const lines = [...new Set(group.map((g) => g.range.start.line + 1))].sort((a, b) => a - b)
474 | 			const lineHint = lines.length > 0 ? ` @ line ${lines.slice(0, 8).join(", ")}${lines.length > 8 ? ", ..." : ""}` : ""
475 | 			const clone = new vscode.Diagnostic(rep.range, `${rep.message} (×${group.length}${lineHint})`, rep.severity)
476 | 			clone.code = rep.code
477 | 			clone.source = rep.source
478 | 			clone.relatedInformation = rep.relatedInformation
479 | 			clone.tags = rep.tags
480 | 			const repUri = DIAGNOSTIC_URI_MAP.get(rep)

481 | 			if (repUri) DIAGNOSTIC_URI_MAP.set(clone, repUri)
482 | 			sampled.push(clone)
483 | 		}
484 | 	}
485 | 
486 | 	return { sampled, total, omitted: total - covered }
487 | }

1206 | function levenshteinDistance(a: string, b: string): number {
1207 | 	if (a === b) return 0
1208 | 	if (a.length === 0) return b.length
1209 | 	if (b.length === 0) return a.length
1210 | 	const row = new Array(b.length + 1)
1211 | 	for (let j = 0; j <= b.length; j++) row[j] = j
1212 | 	for (let i = 1; i <= a.length; i++) {
1213 | 		let prev = i - 1
1214 | 		row[0] = i
1215 | 		for (let j = 1; j <= b.length; j++) {
1216 | 			const tmp = row[j]
1217 | 			const cost = a[i - 1] === b[j - 1] ? 0 : 1
1218 | 			row[j] = Math.min(row[j] + 1, row[j - 1] + 1, prev + cost)
1219 | 			prev = tmp
1220 | 		}
1221 | 	}
1222 | 	return row[b.length]
1223 | }

1226 | 	const maxLen = Math.max(a.length, b.length)
1227 | 	if (maxLen === 0) return 1

1229 | 	if (lenDiff / maxLen > 0.4) return 0
